```javascript
export function ImageWithSkeleton() {
  const [hasLoaded, setHasLoaded] = useState(false);

  return (
    <Skeleton isLoaded={hasLoaded} boxSize="200px">
      <Image
        src="https://placekitten.com/200/200"
        onLoad={() => setHasLoaded(true)}
        alt="A cute kitten"
      />
    </Skeleton>
  );
}
```
